name: CI

on:
  push:
    branches: ["main"]
  pull_request:

jobs:
  frontend:
    name: Frontend (React/Vite)
    runs-on: ubuntu-latest
    if: |
      github.event_name == 'pull_request' ||
      contains(join(github.event.head_commit.modified, ' '), 'frontend/') ||
      contains(join(github.event.head_commit.added, ' '), 'frontend/') ||
      contains(join(github.event.head_commit.removed, ' '), 'frontend/')
    defaults:
      run:
        working-directory: frontend
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
          cache-dependency-path: frontend/package-lock.json

      - name: Install deps
        run: npm ci

      - name: Run tests (optional)
        run: npm run test --if-present

      - name: Build
        run: npm run build

  backend:
    name: Backend (Flask/Python)
    runs-on: ubuntu-latest
    if: |
      github.event_name == 'pull_request' ||
      contains(join(github.event.head_commit.modified, ' '), 'backend/') ||
      contains(join(github.event.head_commit.added, ' '), 'backend/') ||
      contains(join(github.event.head_commit.removed, ' '), 'backend/')
    defaults:
      run:
        working-directory: backend
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"
          cache: pip
          cache-dependency-path: backend/requirements.txt

      - name: Install deps
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt
          pip install pytest

      - name: Run pytest (optional)
        run: pytest -q || echo "No tests yet (optional)"
